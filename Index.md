## Код Jumbo Assembly

```asm
.main:
	; ---------- Тело функции ----------
	dx mov ax, "*", ""
	; -------- Очистка страницы --------
	db mov body, dx mov bh, "body"
	; ---- Создание объекта <body/> ----
	dx mov by, "*", body
	; ---- Рендеринг объекта -----------
	dx mov ax, "body", "Hello, World!"
	; ---------- Вывод текста ----------
end
;-------------- Конец функции --------------

jmp main, ()
;-------------- Вызов функции --------------
```
